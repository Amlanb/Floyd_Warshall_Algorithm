In [9]:
# Construct graph
airport_graph = nx.Graph()
for idx, row in sampled_airports.iterrows():
    airport_graph.add_node(row["ident"], pos=(row["latitude_deg"], row["longitude_deg"]))
for i, j in pairs:
    airport1, airport2 = airport_ids[i], airport_ids[j]
    distance = geodesic(coords[i], coords[j]).km
    airport_graph.add_edge(airport1, airport2, weight=distance)
